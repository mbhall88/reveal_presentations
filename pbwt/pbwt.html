Algorithm 2 - Build prefix and divergent arrays. Update $a$ and $d$ to $a_{k+1}$ and $d_{k+1}$ 

u,v = 0
p,q = k+1
a,b,d,e = []
for i in range(M):
	if d_k[i] > p:
		p = d_k[i]
	if d_k[i] > q:
		q = d_k[i]
	if y[i][k] == 0:
		a[u] = a_k[i]
		d[u] = p
		u += 1
		p = 0
	elif y[i][k] == 1:
		b[v] = a_k[i]
		e[v] = q
		v += 1
		q = 0
return a + b, d + e
